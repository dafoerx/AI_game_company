const dataGenerator = {

  },

  // 生成员工数据
  generateEmployee: (type, specialty, level = 1) => {
    const id = `${type}_${Date.now()}_${Math.random().toString(36).substr(2, 9)}`;
    return {
      id,
      name: `${type === 'human' ? '真人' : 'AI'}员工`,
      type,
      level,
      specialty,
      current_task: null,
      fatigue: 0,
      salary: type === 'human' ? 2000 : 1000
    };
  },

  // 生成项目数据
  generateProject: (size, type = 'mobile_game') => {
    const configs = {
      small: { expected_revenue: 5000, compute_cost: 25, duration: 3 },
      medium: { expected_revenue: 15000, compute_cost: 50, duration: 7 },
      large: { expected_revenue: 50000, compute_cost: 100, duration: 14 }

    return {
      id: `proj_${size}_${Date.now()}`,
      name: `${size === 'small' ? '小型' : size === 'medium' ? '中型' : '大型'}${type === 'mobile_game' ? '手游' : '游戏'}`,

      size,
      progress: 0,
      expected_revenue: config.expected_revenue,
      compute_cost: config.compute_cost,
      start_date: new Date().toISOString().split('T')[0],
      estimated_end: endDate.toISOString().split('T')[0],
      assigned_employees: []
    };
  }
};

function generateBatchData(count, template) {
  const batch = [];

## 十、数据验证规则

### 10.1 数据完整性验证

```javascript
const validationRules = {
  company: {
    required: ['id', 'name', 'cash', 'compute_power', 'morale'],
    ranges: {
      cash: { min: -10000, max: 1000000 },
      compute_power: { min: 0, max: 10000 },
      morale: { min: 0, max: 100 },
      reputation: { min: 0 }
    },
    dependencies: {
      compute_power: { max: 'compute_max' }
    }
  },
  
  employees: {
    required: ['id', 'name', 'type', 'specialty'],
    allowedValues: {
      type: ['human', 'ai'],
      specialty: ['programming', 'art', 'design', 'sound', 'management']
    },
    ranges: {
      level: { min: 1, max: 10 },
      fatigue: { min: 0, max: 100 },
      salary: { min: 0, max: 10000 }
    }
  },
  
  projects: {
    required: ['id', 'name', 'type', 'size', 'progress'],
    allowedValues: {
      size: ['small', 'medium', 'large'],
      type: ['mobile_game', 'pc_game', 'console_game']
    },
    ranges: {
      progress: { min: 0, max: 100 },
      expected_revenue: { min: 0 },
      compute_cost: { min: 0 }
    }
  }
};

// 验证函数
function validateData(data, rules) {
  const errors = [];
  
  // 验证公司数据
  if (data.company) {
    const company = data.company;
    // 检查必填字段
    rules.company.required.forEach(field => {
      if (company[field] === undefined || company[field] === null) {
        errors.push(`公司缺少必填字段: ${field}`);
      }
    });
    
    // 检查数值范围
    Object.entries(rules.company.ranges).forEach(([field, range]) => {
      if (company[field] !== undefined) {
        if (company[field] < range.min || company[field] > range.max) {
          errors.push(`公司字段 ${field} 超出范围: ${company[field]} (允许范围: ${range.min}-${range.max})`);
        }
      }
    });
  }
  
  // 类似地验证员工和项目数据...
  
  return {
    isValid: errors.length === 0,
    errors
  };
}
```

### 10.2 数据一致性验证

```javascript
// 数据一致性验证
function checkConsistency(data) {
  const inconsistencies = [];
  
  // 检查员工分配一致性
  if (data.employees && data.projects) {
    const assignedEmployeeIds = new Set();
    data.projects.forEach(project => {
      project.assigned_employees.forEach(empId => {
        assignedEmployeeIds.add(empId);
      });
    });
    
    // 检查是否有员工被分配到不存在的项目
    assignedEmployeeIds.forEach(empId => {
      const employeeExists = data.employees.some(emp => emp.id === empId);
      if (!employeeExists) {
        inconsistencies.push(`员工 ${empId} 被分配到项目，但该员工不存在`);
      }
    });
    
    // 检查是否有员工被重复分配
    const employeeAssignments = {};
    data.employees.forEach(emp => {
      employeeAssignments[emp.id] = 0;
    });
    
    data.projects.forEach(project => {
      project.assigned_employees.forEach(empId => {
        if (employeeAssignments[empId] !== undefined) {
          employeeAssignments[empId]++;
        }
      });
    });
    
    Object.entries(employeeAssignments).forEach(([empId, count]) => {
      if (count > 1) {
        inconsistencies.push(`员工 ${empId} 被分配到 ${count} 个项目，可能超负荷`);
      }
    });
  }
  
  // 检查资源一致性
  if (data.company) {
    const totalSalary = data.employees 
      ? data.employees.reduce((sum, emp) => sum + emp.salary, 0)
      : 0;
    
    if (data.company.cash < totalSalary * 0.5) {
      inconsistencies.push(`现金可能不足以支付下个月工资 (现金: ${data.company.cash}, 月薪总额: ${totalSalary})`);
    }
  }
  
  return inconsistencies;
}
```

class DataImporter {
  constructor() {
    this.supportedFormats = ['json', 'csv'];
  }
  
  importJSON(dataString) {
    try {
      const data = JSON.parse(dataString);
      
      // 验证数据格式
      const validation = validateData(data, validationRules);
      if (!validation.isValid) {
        throw new Error(`数据验证失败: ${validation.errors.join(', ')}`);
      }
      
      // 检查一致性
      const inconsistencies = checkConsistency(data);
      if (inconsistencies.length > 0) {
        console.warn('数据不一致警告:', inconsistencies);
      }
      
      return {
        success: true,
        data,
        warnings: inconsistencies
      };
    } catch (error) {
      return {
        success: false,
        error: error.message
      };
    }
  }
  
  importCSV(csvString, mapping) {
    // CSV导入实现
    // ...
  }
  
  importFile(file, format) {
    if (!this.supportedFormats.includes(format)) {
      return {
        success: false,
        error: `不支持的文件格式: ${format}`
      };
    }
    
    const reader = new FileReader();
    return new Promise((resolve, reject) => {
      reader.onload = (event) => {
        const content = event.target.result;
        let result;
        
        if (format === 'json') {
          result = this.importJSON(content);
        } else if (format === 'csv') {
          result = this.importCSV(content);
        }
        
        resolve(result);
      };
      
      reader.onerror = () => {
        reject(new Error('文件读取失败'));
      };
      
      reader.readAsText(file);
    });
  }
}

        source: '测试数据导出'
      },
      data
    };
    
    if (options.pretty) {
      return JSON.stringify(exportData, null, 2);
    }
    return JSON.stringify(exportData);
  }
  
  exportCSV(data, options = {}) {
    // CSV导出实现
    // ...
  }
  
  exportData(data, format, options = {}) {
    const exporter = this.exportFormats[format];
    if (!exporter) {
      throw new Error(`不支持的导出格式: ${format}`);
    }
    
    return exporter(data, options);
  }
  
  download(data, filename, format) {
    const content = this.exportData(data, format);
    const blob = new Blob([content], { type: 'text/plain' });
    const url = URL.createObjectURL(blob);
    
    const a = document.createElement('a');
    a.href = url;
    a.download = filename;
    document.body.appendChild(a);
    a.click();
    document.body.removeChild(a);
    URL.revokeObjectURL(url);
  }
